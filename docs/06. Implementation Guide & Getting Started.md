## 3. Key Component Implementations

### 3.1 Offline Manager

```typescript
// /src/hooks/use-offline.ts
import { useState, useEffect } from 'react'
import { zustandOfflineStore } from '@/stores/offline-store'

export function useOffline() {
  const [isOnline, setIsOnline] = useState(true)
  const { queueAction, syncQueuedActions } = zustandOfflineStore()

  useEffect(() => {
    const handleOnline = () => {
      setIsOnline(true)
      syncQueuedActions() // Sync when back online
    }
    
    const handleOffline = () => setIsOnline(false)

    window.addEventListener('online', handleOnline)
    window.addEventListener('offline', handleOffline)

    return () => {
      window.removeEventListener('online', handleOnline)
      window.removeEventListener('offline', handleOffline)
    }
  }, [])

  return {
    isOnline,
    queueAction,
    syncQueuedActions,
  }
}
```

interface CourseDownloadProps {
  courseId: string
  courseSize: number
  canDownload: boolean
}

export function CourseDownload({ 
  courseId, 
  courseSize, 
  canDownload 
}: CourseDownloadProps) {
  const { isOnline, queueAction } = useOffline()
  const [isDownloading, setIsDownloading] = useState(false)

  const handleDownload = async () => {
    if (!canDownload) return
    
    setIsDownloading(true)

        type: 'DOWNLOAD_COURSE',
        payload: { courseId },
        timestamp: new Date().toISOString(),

      console.error('Download failed:', error)
    } finally {
      setIsDownloading(false)
    }
  }

      <Button

        disabled={!canDownload || !isOnline || isDownloading}

## 4. API Route Implementations

### 4.1 Course Validation API

```typescript
// /src/app/api/courses/[id]/validate/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { auth } from '@/lib/auth'
import { prisma } from '@/lib/db'

export async function POST(
  request: NextRequest,
  { params }: { params: { id: string } }
) {
  try {
    const session = await auth()
    if (!session || !['ADMIN', 'RESPONSABLE_PEDAGOGIQUE'].includes(session.user.role)) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 403 })
    }

    const { status, feedback } = await request.json()
    const courseId = params.id

    // Update course status
    const course = await prisma.course.update({
      where: { id: courseId },
      data: { 
        status: status === 'approved' ? 'PUBLISHED' : 'DRAFT',
        lastValidatedAt: new Date(),
        validatedById: session.user.id,
      },
    })

    // Add validation record
    await prisma.courseValidation.create({
      data: {
        courseId,
        reviewerId: session.user.id,
        status: status.toUpperCase(),
        feedbackText: feedback,
        reviewedAt: new Date(),
      },
    })

    return NextResponse.json({ course })
  } catch (error) {
    console.error('Validation error:', error)
    return NextResponse.json({ error: 'Internal server error' }, { status: 500 })
  }
}
```

import { NextRequest, NextResponse } from 'next/server'
import { auth } from '@/lib/auth'
import { prisma } from '@/lib/db'

export async function POST(request: NextRequest) {
  try {

    })
  } catch (error) {
    console.error('Sync error:', error)
    return NextResponse.json({ error: 'Sync failed' }, { status: 500 })
  }
}

## 5. State Management with Zustand

### 5.1 Offline Store

```typescript
// /src/stores/offline-store.ts
import { create } from 'zustand'
import { persist } from 'zustand/middleware'

interface OfflineAction {
  type: string
  payload: any
  timestamp: string
  id: string
}

interface OfflineStore {
  queuedActions: OfflineAction[]
  cachedCourses: Record<string, any>
  lastSync: string | null
  queueAction: (action: Omit<OfflineAction, 'id'>) => void
  syncQueuedActions: () => Promise<void>
  cacheCourse: (courseId: string, courseData: any) => void
  getCachedCourse: (courseId: string) => any
}

export const useOfflineStore = create<OfflineStore>()(
  persist(
    (set, get) => ({
      queuedActions: [],
      cachedCourses: {},
      lastSync: null,

      queueAction: (action) => {
        const newAction = {
          ...action,
          id: Math.random().toString(36).substr(2, 9),
        }
        
        set((state) => ({
          queuedActions: [...state.queuedActions, newAction],
        }))
      },

      syncQueuedActions: async () => {
        const { queuedActions } = get()
        
        for (const action of queuedActions) {
          try {
            await fetch('/api/sync/push', {
              method: 'POST',
              body: JSON.stringify({ actions: [action] }),
            })
          } catch (error) {
            console.error('Failed to sync action:', error)
          }
        }

        // Clear successfully synced actions
        set({ queuedActions: [] })
      },

      cacheCourse: (courseId, courseData) => {
        set((state) => ({
          cachedCourses: {
            ...state.cachedCourses,
            [courseId]: {
              ...courseData,
              cachedAt: new Date().toISOString(),
            },
          },
        }))
      },

      getCachedCourse: (courseId) => {
        const { cachedCourses } = get()
        return cachedCourses[courseId]
      },
    }),
    {
      name: 'offline-storage',
      // Use IndexedDB or localStorage for persistence
    }
  )
)
```

// /src/app/[lang]/(dashboard)/page.tsx
import { getCurrentUser } from '@/lib/auth'
import { TeacherDashboard } from '@/components/dashboard/teacher-dashboard'
import { StudentDashboard } from '@/components/dashboard/student-dashboard'
import { ParentDashboard } from '@/components/dashboard/parent-dashboard'
import { AdminDashboard } from '@/components/dashboard/admin-dashboard'

export default async function DashboardPage() {
  const user = await getCurrentUser()
  
  const renderDashboard = () => {
    switch (user.role) {
      case 'TEACHER':
        return <TeacherDashboard user={user} />
      case 'STUDENT':
        return <StudentDashboard user={user} />
      case 'PARENT':
        return <ParentDashboard user={user} />
      case 'ADMIN':
      case 'RESPONSABLE_PEDAGOGIQUE':
        return <AdminDashboard user={user} />
      default:
        return <div>Dashboard non disponible</div>
    }
  }

  return (
    <div className="space-y-6">
      <div className="flex items-center justify-between">
        <h1 className="text-3xl font-bold">
          {user.role === 'TEACHER' && 'Tableau de Bord Enseignant'}
          {user.role === 'STUDENT' && 'Mon Tableau de Bord'}
          {user.role === 'PARENT' && 'Tableau de Bord Parent'}
          {user.role === 'ADMIN' && 'Administration'}
        </h1>
      </div>
      {renderDashboard()}
    </div>
  )
}
